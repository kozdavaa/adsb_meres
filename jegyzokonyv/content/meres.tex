\begin{lstlisting}
// Decoding
if (fifo[i] > (accumulator/FIR_LEN))
    bit = 1;
else
	bit = 0;
	// ADS-B packet search and print
	if (stm < 16)
	{
		if(bit == adsb_preamble[stm])
			stm++;
        else
			stm = 0;
    }
	else if((stm>=16)&&(stm<PCKT_LEN))
	{
		if (stm==16) printf("*");
		if ((stm%2)==0)
		{
			printf("%d",bit);
			hex=hex|bit;
			j++;
			if(j==8) {
                printf("%02x",hex);
				hex = 0;
				j = 0;
			}
			else hex = hex<<1;
		}
            stm++;
	}
	else
	{
		printf(";\r\n");
		stm = 0;
	}
\end{lstlisting}
